```js
const express = require('express');
const bcrypt = require('bcrypt');
const jwt = require('jsonwebtoken');
const { getDB } = require('../db/db');
const { logger } = require('../logging');

const router = express.Router();
const JWT_SECRET = process.env.JWT_SECRET || 'secret';
const JWT_EXPIRY = process.env.JWT_EXPIRY || '1h';

router.post('/register', async (req, res, next) => {
  try {
    const db = getDB();
    const { username, password } = req.body;
    if (!username || !password) return res.status(400).json({ error: 'username & password required' });
    const password_hash = await bcrypt.hash(password, 10);
    const stmt = await db.run('INSERT INTO users (username, password_hash) VALUES (?, ?)', [username, password_hash]);
    logger.info('User registered', { userId: stmt.lastID });
    res.status(201).json({ id: stmt.lastID, username });
  } catch (err) {
    if (err && err.code === 'SQLITE_CONSTRAINT') return res.status(409).json({ error: 'username_taken' });
    next(err);
  }
});

router.post('/login', async (req, res, next) => {
  try {
    const db = getDB();
    const { username, password } = req.body;
    if (!username || !password) return res.status(400).json({ error: 'username & password required' });
    const user = await db.get('SELECT * FROM users WHERE username = ?', [username]);
    if (!user) return res.status(401).json({ error: 'invalid_credentials' });
    const ok = await bcrypt.compare(password, user.password_hash);
    if (!ok) return res.status(401).json({ error: 'invalid_credentials' });
    const token = jwt.sign({ sub: user.id, username }, JWT_SECRET, { expiresIn: JWT_EXPIRY });
    res.json({ token, user: { id: user.id, username } });
  } catch (err) { next(err); }
});

module.exports = router;
```

```js
const jwt = require('jsonwebtoken');
const JWT_SECRET = process.env.JWT_SECRET || 'secret';

function authenticate(req, res, next) {
  const auth = req.headers.authorization;
  if (!auth || !auth.startsWith('Bearer ')) return res.status(401).json({ error: 'missing_token' });
  const token = auth.slice(7);
  try {
    const payload = jwt.verify(token, JWT_SECRET);
    req.user = payload;
    next();
  } catch (err) {
    return res.status(401).json({ error: 'invalid_token' });
  }
}

module.exports = { authenticate };
```

src/models/probability.js
```js
function impliedProbabilityFromDecimal(decimalOdds) {
  if (!decimalOdds || decimalOdds <= 1) return null;
  return 1 / decimalOdds;
}

function calibrateProbability(rawP) {
  if (rawP === null || rawP === undefined) return 0.5;
  const alpha = 0.9;
  return alpha * rawP + (1 - alpha) * 0.5;
}

function estimateLegProbability(leg) {
  if (leg.probOverride !== undefined) return Math.max(0.001, Math.min(0.999, leg.probOverride));
  if (leg.decimalOdds) {
    const p = impliedProbabilityFromDecimal(leg.decimalOdds);
    return calibrateProbability(p);
  }
  return 0.5;
}

module.exports = { estimateLegProbability };
```

src/models/correlation.js
```js
function clamp(x, lo = -0.99, hi = 0.99) {
  return Math.max(lo, Math.min(hi, x));
}

function baseCorrelationForMarkets(a, b) {
  if (a.market === b.market && a.market === 'moneyline') return 0.6;
  if (a.market === b.market && a.market === 'spread') return 0.5;
  if (a.market === b.market && a.market === 'total') return 0.4;
  if ((a.market === 'moneyline' && b.market === 'spread') || (b.market === 'moneyline' && a.market === 'spread')) return 0.45;
  return 0.15;
}

function estimatePairCorrelation(a, b) {
  if (!a || !b) return 0;
  if (a.eventId && b.eventId && a.eventId === b.eventId) {
    if (a.outcome === b.outcome) {
      return clamp(0.6 + 0.2 * (baseCorrelationForMarkets(a, b)));
    }
    return clamp(-0.85);
  }

  const teamOverlap = (a.team && b.team && a.team === b.team) || (a.home_team && b.home_team && a.home_team === b.home_team);
  if (teamOverlap) {
    return clamp(0.25);
  }

  const base = baseCorrelationForMarkets(a, b);
  return clamp(base * 0.2);
}

function buildCorrelationMatrix(legs) {
  const n = legs.length;
  const mat = Array.from({ length: n }, () => Array(n).fill(0));
  for (let i = 0; i < n; i++) {
    for (let j = 0; j < n; j++) {
      if (i === j) mat[i][j] = 1;
      else mat[i][j] = estimatePairCorrelation(legs[i], legs[j]);
    }
  }
  const eps = 1e-6;
  for (let i = 0; i < n; i++) mat[i][i] += eps;
  return mat;
}

module.exports = { buildCorrelationMatrix, estimatePairCorrelation };
```

src/montecarlo/engine.js
```js
const { estimateLegProbability } = require('../models/probability');
const { buildCorrelationMatrix } = require('../models/correlation');
const { logger } = require('../logging');
const math = require('mathjs');

function cholDecomp(A) {
  const n = A.length;
  const L = Array.from({ length: n }, () => Array(n).fill(0));
  for (let i = 0; i < n; i++) {
    for (let j = 0; j <= i; j++) {
      let s = 0;
      for (let k = 0; k < j; k++) s += L[i][k] * L[j][k];
      if (i === j) {
        const val = A[i][i] - s;
        L[i][j] = Math.sqrt(Math.max(val, 1e-12));
      } else {
        L[i][j] = (A[i][j] - s) / L[j][j];
      }
    }
  }
  return L;
}

function gaussianSampleVector(n) {
  const out = Array(n).fill(0);
  for (let i = 0; i < n; i += 2) {
    const u1 = Math.random() || 1e-12;
    const u2 = Math.random() || 1e-12;
    const z0 = Math.sqrt(-2.0 * Math.log(u1)) * Math.cos(2.0 * Math.PI * u2);
    const z1 = Math.sqrt(-2.0 * Math.log(u1)) * Math.sin(2.0 * Math.PI * u2);
    out[i] = z0;
    if (i + 1 < n) out[i + 1] = z1;
  }
  return out;
}

function normInv(p) {
  if (p <= 0) return -Infinity;
  if (p >= 1) return Infinity;
  return Math.sqrt(2) * math.erfinv(2 * p - 1);
}

async function evaluateParlay(legs, { sims = 20000 } = {}) {
  const n = legs.length;
  const probs = legs.map(estimateLegProbability);
  const corr = buildCorrelationMatrix(legs);

  let independent = true;
  for (let i = 0; i < n && independent; i++) {
    for (let j = 0; j < n; j++) {
      if (i !== j && Math.abs(corr[i][j]) > 1e-3) { independent = false; break; }
    }
  }
  if (independent) {
    const winProb = probs.reduce((acc, p) => acc * p, 1);
    return { winProb, method: 'analytic', sims: 0 };
  }

  const thresholds = probs.map(p => normInv(p));
  const L = cholDecomp(corr);

  let wins = 0;
  const batchSize = Math.min(5000, sims);
  let run = 0;
  while (run < sims) {
    const thisBatch = Math.min(batchSize, sims - run);
    for (let b = 0; b < thisBatch; b++) {
      const z = gaussianSampleVector(n);
      const x = Array(n).fill(0);
      for (let i = 0; i < n; i++) {
        let acc = 0;
        for (let k = 0; k <= i; k++) acc += L[i][k] * z[k];
        x[i] = acc;
      }
      let ok = true;
      for (let i = 0; i < n; i++) {
        if (x[i] > thresholds[i]) { ok = false; break; }
      }
      if (ok) wins++;
    }
    run += thisBatch;
    await new Promise(r => setTimeout(r, 0));
  }

  const winProb = wins / sims;
  logger.info('MonteCarlo done', { sims, winProb });
  return { winProb, method: 'montecarlo', sims };
}

module.exports = { evaluateParlay };
```

src/optimizer/beamSearch.js
```js
const { estimateLegProbability } = require('../models/probability');
const { evaluateParlay } = require('../montecarlo/engine');
const { logger } = require('../logging');

async function findBestParlay(candidates, { maxLegs = 6, beamWidth = 100 } = {}) {
  let beam = [{ legs: [], analytic: 1 }];
  for (let depth = 1; depth <= maxLegs; depth++) {
    const expansions = [];
    for (const state of beam) {
      for (const cand of candidates) {
        if (state.legs.find(l => l.id === cand.id)) continue;
        const newLegs = state.legs.concat(cand);
        const probs = newLegs.map(estimateLegProbability);
        const analytic = probs.reduce((a, b) => a * b, 1);
        expansions.push({ legs: newLegs, analytic });
      }
    }
    expansions.sort((a, b) => b.analytic - a.analytic);
    beam = expansions.slice(0, beamWidth);
    if (beam.length === 0) break;
  }

  const scored = [];
  for (const b of beam.slice(0, 20)) {
    const mc = await evaluateParlay(b.legs, { sims: 5000 });
    scored.push({ legs: b.legs, analytic: b.analytic, montecarlo: mc.winProb });
  }
  scored.sort((a, b) => b.montecarlo - a.montecarlo);
  logger.info('Optimizer finished', { top: scored.length });
  return scored.slice(0, 10);
}

module.exports = { findBestParlay };
```

src/utils/kelly.js
```js
function kellyFraction(p, decimalOdds) {
  if (!p || !decimalOdds || decimalOdds <= 1) return 0;
  const b = decimalOdds - 1;
  const f = (p * (decimalOdds) - 1) / b;
  return Math.max(0, f);
}

function fractionalKelly(p, decimalOdds, fraction = 0.5) {
  return Math.max(0, Math.min(1, fraction * kellyFraction(p, decimalOdds)));
}

module.exports = { kellyFraction, fractionalKelly };
```

src/backtest/engine.js
```js
const { estimateLegProbability } = require('../models/probability');

async function runBacktest(historicalParlays = [], strategy = {}) {
  let bankroll = strategy.bankroll || 1000;
  const initial = bankroll;
  let peak = bankroll;
  let maxDrawdown = 0;
  let wins = 0;
  for (const item of historicalParlays) {
    const legs = item.legs;
    const probs = legs.map(estimateLegProbability);
    const p = probs.reduce((a, b) => a * b, 1);
    let stake = 0;
    if (strategy.stakeSizing?.type === 'fixed') {
      stake = strategy.stakeSizing.amount;
    } else if (strategy.stakeSizing?.type === 'kelly') {
      const decimalOdds = item.payoutDecimal || legs.reduce((acc, l) => acc * (l.decimalOdds || 2), 1);
      const f = require('../utils/kelly').kellyFraction(p, decimalOdds);
      stake = bankroll * f;
    } else if (strategy.stakeSizing?.type === 'fractionalKelly') {
      const decimalOdds = item.payoutDecimal || legs.reduce((acc, l) => acc * (l.decimalOdds || 2), 1);
      const frac = strategy.stakeSizing.fraction || 0.5;
      const f = require('../utils/kelly').fractionalKelly(p, decimalOdds, frac);
      stake = bankroll * f;
    } else {
      stake = bankroll * 0.01;
    }
    stake = Math.max(0, Math.min(bankroll, stake));
    if (item.result === 1) {
      const payout = (item.payoutDecimal || legs.reduce((acc, l) => acc * (l.decimalOdds || 2), 1)) * stake;
      bankroll = bankroll - stake + payout;
      wins++;
    } else {
      bankroll = bankroll - stake;
    }
    if (bankroll > peak) peak = bankroll;
    maxDrawdown = Math.max(maxDrawdown, (peak - bankroll) / peak);
  }
  const roi = (bankroll - initial) / initial;
  return { initialBankroll: initial, finalBankroll: bankroll, roi, maxDrawdown, winRate: historicalParlays.length ? wins / historicalParlays.length : 0 };
}

module.exports = { runBacktest };
```

```js
const { logger } = require('../logging');

async function withRetries(fn, { retries = 3, backoffMs = 200 } = {}) {
  let attempt = 0;
  while (attempt <= retries) {
    try {
      return await fn();
    } catch (err) {
      attempt++;
      logger.warn('Retry attempt failed', { attempt, err: err.message });
      if (attempt > retries) throw err;
      await new Promise(r => setTimeout(r, backoffMs * attempt));
    }
  }
}

module.exports = { withRetries };
```

```js
    const top = await optimizer.findBestParlay(candidates, { maxLegs, beamWidth });
    const out = await Promise.all(top.map(async (t) => {
      const evalRes = await montecarlo.evaluateParlay(t.legs, { sims: 5000 });
      const decimalOdds = t.legs.reduce((acc, l) => acc * (l.decimalOdds || 2), 1);
      const p = evalRes.winProb;
      const fullKelly = kellyFraction(p, decimalOdds);
      const fracKelly = fractionalKelly(p, decimalOdds, kellyFrac);
```

```js
import React, { useState } from 'react';
import Login from './pages/Login';
import ParlayBuilder from './pages/ParlayBuilder';

export default function App() {
  const [token, setToken] = useState(localStorage.getItem('token') || null);
  if (!token) return <Login onLogin={(t) => { localStorage.setItem('token', t); setToken(t); }} />;
  return <ParlayBuilder token={token} onLogout={() => { localStorage.removeItem('token'); setToken(null); }} />;
}
```

```js
import React, { useState } from 'react';
import axios from 'axios';

export default function Login({ onLogin }) {
  const [username, setUsername] = useState('');
  const [password, setPassword] = useState('');
  const [err, setErr] = useState(null);

  async function submit(e) {
    e.preventDefault();
    setErr(null);
    try {
      const res = await axios.post('/auth/login', { username, password });
      onLogin(res.data.token);
    } catch (e) {
      setErr(e.response?.data?.error || 'login_failed');
    }
  }

  return (
    <div style={{ padding: 20 }}>
      <h2>Login</h2>
      <form onSubmit={submit}>
        <div><input placeholder="username" value={username} onChange={e => setUsername(e.target.value)} /></div>
        <div><input placeholder="password" type="password" value={password} onChange={e => setPassword(e.target.value)} /></div>
        <button type="submit">Login</button>
      </form>
      {err && <div style={{ color: 'red' }}>{err}</div>}
    </div>
  );
}
```

  function addLeg(event) {
    const leg = { id: `${event.id}-${Date.now()}`, eventId: event.id, market: 'moneyline', outcome: event.home_team, decimalOdds: 2.0, home_team: event.home_team, visitor_team: event.visitor_team };
    setSelected(s => [...s, leg]);

  async function evaluate() {
    setEvalResult(null);
    const res = await axios.post('/api/parlay/evaluate', { legs: selected, sims: 5000 }, { headers: { Authorization: `Bearer ${token}` }});
    setEvalResult(res.data);
  }

  async function optimizeAndSize() {
    setOptimizerResults([]);

    setOptimizerResults(res.data);
